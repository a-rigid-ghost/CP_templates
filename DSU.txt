////1-based

int find(int x,lli par[]){
    if(par[x]!=x){
        par[x]=find(par[x],par);
    }
    return par[x];
}
 
void merge(int x,int y,lli par[],lli val[]){
    int xr=find(x,par),yr=find(y,par);
    if(val[xr]<val[yr])swap(xr,yr);
    par[yr]=xr;
    val[xr]+=val[yr];
}
